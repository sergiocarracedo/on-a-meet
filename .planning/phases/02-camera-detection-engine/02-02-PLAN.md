# Plan 02-02: Polling Engine + Detect Command

<objective>
Build the polling engine that drives the V4L2 detector, then wire it to the detect command for real-time terminal output. The engine handles multi-camera OR logic, debounce (N consecutive same-state polls), hotplug detection (re-enumerate each cycle), --camera filtering, and graceful shutdown via context. The detect command shows a startup banner listing all cameras and prints state change lines.

```go
package engine

import (
    "context"
    "log"
    "sync"
    "time"

    "github.com/on-a-meet/internal/detector"
)

type OnChange func(path string, oldState, newState bool, info detector.DeviceInfo)

type deviceState struct {
    info           detector.DeviceInfo
    current        bool
    previous       bool
    debounceCount  int
    debounceTarget int
}

type Engine struct {
    detector     detector.Detector
    interval     time.Duration
    debounce     int
    cameraFilter string
    onChange     OnChange

    states map[string]*deviceState
    mu     sync.Mutex
    logger *log.Logger
}
```

Constructors and options:

```go
type Option func(*Engine)

func WithInterval(d time.Duration) Option {
    return func(e *Engine) { e.interval = d }
}

func WithDebounce(n int) Option {
    return func(e *Engine) { e.debounce = n }
}

func WithCameraFilter(path string) Option {
    return func(e *Engine) { e.cameraFilter = path }
}

func WithOnChange(cb OnChange) Option {
    return func(e *Engine) { e.onChange = cb }
}

func New(det detector.Detector, opts ...Option) *Engine {
    e := &Engine{
        detector: det,
        interval: 1 * time.Second,
        debounce: 3,
        states:   make(map[string]*deviceState),
        logger:   log.New(log.Writer(), "", 0),
    }
    for _, opt := range opts {
        opt(e)
    }
    return e
}
```

Implement `Run(ctx context.Context) error`:

```go
func (e *Engine) Run(ctx context.Context) error {
    // 1. Initial device enumeration
    devices, err := e.detector.ListDevices()
    if err != nil {
        return err
    }

    // Apply --camera filter
    var filtered []detector.DeviceInfo
    for _, d := range devices {
        if e.cameraFilter != "" && d.Path != e.cameraFilter {
            continue
        }
        filtered = append(filtered, d)
    }

    // Initialize states for existing devices
    for _, d := range filtered {
        status, err := e.detector.Detect(d.Path)
        if err != nil {
            // Device disappeared between enumeration and detect — skip
            continue
        }
        e.mu.Lock()
        e.states[d.Path] = &deviceState{
            info:           d,
            current:        status.On,
            previous:       status.On,
            debounceTarget: e.debounce,
        }
        e.mu.Unlock()
    }

    // Poll loop
    for {
        select {
        case <-ctx.Done():
            return ctx.Err()
        case <-time.After(e.interval):
            e.pollCycle()
        }
    }
}
```

Implement `pollCycle()`:

```go
func (e *Engine) pollCycle() {
    // 1. Re-enumerate devices for hotplug detection
    devices, err := e.detector.ListDevices()
    if err != nil {
        return
    }

    // Apply --camera filter
    filtered := make(map[string]detector.DeviceInfo)
    for _, d := range devices {
        if e.cameraFilter != "" && d.Path != e.cameraFilter {
            continue
        }
        filtered[d.Path] = d
    }

    // Detect new devices (hotplug add)
    var added []struct {
        path string
        info detector.DeviceInfo
    }
    for path, info := range filtered {
        if _, exists := e.states[path]; !exists {
            e.states[path] = &deviceState{
                info:           info,
                debounceTarget: e.debounce,
            }
            added = append(added, struct {
                path string
                info detector.DeviceInfo
            }{path, info})
        }
    }
    e.mu.Unlock()
    for _, a := range added {
        if e.onChange != nil {
            e.onChange(a.path, false, false, a.info)
        }
    }

    // Detect removed devices (hotplug remove)
    var removed []struct {
        path string
        info detector.DeviceInfo
    }
    for path := range e.states {
        if _, exists := filtered[path]; !exists {
            removed = append(removed, struct {
                path string
                info detector.DeviceInfo
            }{path, e.states[path].info})
            delete(e.states, path)
        }
    }
    e.mu.Unlock()
    for _, r := range removed {
        if e.onChange != nil {
            e.onChange(r.path, true, false, r.info)
        }
    }

    // 2. Poll each device (no lock needed — we only access filtered map iteratively)
    for path := range filtered {
        status, err := e.detector.Detect(path)
        if err != nil {
            continue
        }

        e.mu.Lock()
        s := e.states[path]
        e.mu.Unlock()

        if s == nil {
            continue
        }

        // Debounce: N consecutive same NEW readings before accepting change
        if status.On == s.current {
            // No change — reset debounce counter
            s.debounceCount = 0
        } else {
            // Potential change — increment, fire only at threshold
            s.debounceCount++
            if s.debounceCount >= s.debounceTarget {
                oldState := s.current
                s.current = status.On
                s.previous = oldState
                s.debounceCount = 0
                if e.onChange != nil {
                    e.onChange(path, oldState, status.On, s.info)
                }
            }
        }
    }
}
```

Handle hotplug remove callback correctly (rework: track removed devices separately and dispatch after releasing lock).

Important implementation notes:
- Use `e.mu` carefully to avoid deadlocks. The OnChange callback is called outside the lock.
- Handle the race where a new device is added between ListDevices and Detect for that device.
- Use `log.New(log.Writer(), "", 0)` for the logger so pterm output is clean.

```go
RunE: func(cmd *cobra.Command, args []string) error {
    cfg := configFromViper() // use viper to get camera, interval, debounce

    // Parse interval
    interval, err := time.ParseDuration(cfg.Interval)
    if err != nil {
        return err
    }

    // Create V4L2 detector
    det := detector.NewV4L2Detector()

    // Check device accessibility at startup
    devices, err := det.ListDevices()
    if err != nil {
        output.Error.Println("Failed to enumerate camera devices:", err)
        return err
    }
    if len(devices) == 0 {
        output.Warning.Println("No camera devices detected. Make sure your camera is connected and you have the right permissions.")
        output.Info.Println("Tip: add your user to the 'video' group: sudo usermod -a -G video $USER")
        output.Info.Println("Then log out and back in, or run: newgrp video")
        return nil
    }

    // Show startup banner with detected cameras
    output.Banner(len(devices))
    for _, d := range devices {
        output.Info.Printfln("  %s — %s (driver: %s)", d.Path, d.Card, d.Driver)
    }

    // Create engine with options
    eng := engine.New(det,
        engine.WithInterval(interval),
        engine.WithDebounce(cfg.Debounce),
        engine.WithOnChange(func(path string, oldState, newState bool, info detector.DeviceInfo) {
            if newState {
                output.Warning.Printfln("%s ⟶ ON  (driver: %s)", path, info.Driver)
            } else {
                output.Info.Printfln("%s ⟶ OFF  (driver: %s)", path, info.Driver)
            }
        }),
    )

    if cfg.Camera != "" {
        output.Info.Printfln("Monitoring only: %s", cfg.Camera)
        eng = engine.WithCameraFilter(cfg.Camera)(eng)
    }

    // Start engine with signal-aware context
    ctx, cancel := context.WithCancel(context.Background())
    defer cancel()

    sigCh := make(chan os.Signal, 1)
    signal.Notify(sigCh, syscall.SIGINT, syscall.SIGTERM)
    go func() {
        <-sigCh
        output.Info.Println("Shutting down...")
        cancel()
    }()

    return eng.Run(ctx)
}
```
