<action>
Create `internal/executor/executor.go`:

```go
package executor

import (
    "bytes"
    "context"
    "fmt"
    "os/exec"
    "sync"
    "syscall"
    "text/template"
    "time"
)

// TemplateData holds the variables available in command templates.
type TemplateData struct {
    CameraID string // short device name, e.g. "video0"
    Device   string // full device path, e.g. "/dev/video0"
    State    string // "on" or "off"
}

// Executor runs user-defined commands with template substitution.
type Executor struct {
    timeout time.Duration
    running sync.Map // key: "on" or "off", value: context.CancelFunc
}

// New creates an Executor with the given timeout.
// If timeout is 0, commands have no time limit.
func New(timeout time.Duration) *Executor {
    return &Executor{timeout: timeout}
}

// ExecOn runs the on-command for the given camera. Skips if another on-command
// is already running for the same session.
func (e *Executor) ExecOn(ctx context.Context, cmdStr string, data TemplateData) error {
    return e.exec(ctx, cmdStr, data, "on")
}

// ExecOff runs the off-command for the given camera. Skips if another off-command
// is already running for the same session.
func (e *Executor) ExecOff(ctx context.Context, cmdStr string, data TemplateData) error {
    return e.exec(ctx, cmdStr, data, "off")
}

func (e *Executor) exec(ctx context.Context, cmdStr string, data TemplateData, state string) error {
    // Check if a command for this state is already running
    if _, loaded := e.running.Load(state); loaded {
        return nil // skip — same-state already in flight
    }

    // Create per-command context
    cmdCtx := ctx
    var cancel context.CancelFunc
    if e.timeout > 0 {
        cmdCtx, cancel = context.WithTimeout(ctx, e.timeout)
    } else {
        cmdCtx, cancel = context.WithCancel(ctx)
    }
    defer cancel()

    // Store the cancel func in the running map
    e.running.Store(state, cancel)
    defer e.running.Delete(state)

    // Render template
    tmpl, err := template.New("").Parse(cmdStr)
    if err != nil {
        return fmt.Errorf("template parse error: %w", err)
    }
    var buf bytes.Buffer
    if err := tmpl.Execute(&buf, data); err != nil {
        return fmt.Errorf("template execute error: %w", err)
    }
    rendered := buf.String()

    // Build shell command
    cmd := exec.CommandContext(cmdCtx, "sh", "-c", rendered)
    cmd.SysProcAttr = &syscall.SysProcAttr{Setpgid: true}

    // Go 1.20+: kill process group on context cancellation
    cmd.Cancel = func() error {
        if cmd.Process != nil && cmd.Process.Pid > 0 {
            return syscall.Kill(-cmd.Process.Pid, syscall.SIGKILL)
        }
        return nil
    }
    cmd.WaitDelay = 5 * time.Second

    // Capture combined output
    var outBuf bytes.Buffer
    cmd.Stdout = &outBuf
    cmd.Stderr = &outBuf

    // Run
    err = cmd.Run()
    output := outBuf.String()

    if err != nil {
        if ctx.Err() != nil {
            return fmt.Errorf("command timed out after %v: %s", e.timeout, output)
        }
        return fmt.Errorf("command failed (exit code %d): %s", cmd.ProcessState.ExitCode(), output)
    }

    return nil
}
```

Note the key design decisions:
- `sync.Map` tracks running commands per state ("on"/"off") — this enables same-state skip
- `defer e.running.Delete(state)` ensures cleanup when command finishes or times out
- Process group (`Setpgid: true`) + `syscall.Kill(-pid, SIGKILL)` ensures children are killed on timeout
- `WaitDelay` provides a grace period after cancel before force-kill
- Combined stdout/stderr captured for error reporting
